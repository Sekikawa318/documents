def read_s3_csv(SRC_OBJECT_KEY_NAME, SRC_FILE_ENCODING="utf-8", SRC_BUCKET_NAME="dataanalytics-training-2021"):
    """
    S3csvpandasdf

    Parameters
    ----------
    SRC_BUCKET_NAME : str
        S3
        dataanalytics-training-2021
        
    SRC_OBJECT_KEY_NAME : str
        csv
    SRC_FILE_ENCODINGstr
        csv
        
    Returns
    -------
   df : pandas.DataFrame
        csv
    """

    # 
    s3 = boto3.resource('s3')
    src_obj = s3.Object(SRC_BUCKET_NAME, SRC_OBJECT_KEY_NAME)
    body_in = src_obj.get()['Body'].read().decode(SRC_FILE_ENCODING)

    # 
    buffer_in = io.StringIO(body_in)

    # 
    df = pd.read_csv(buffer_in)
    
    return df

train_df = read_s3_csv('train.csv')
